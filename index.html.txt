      const [time, setTime] = useState({ hours: 0, minutes: 0, seconds: 0 });
      const [isChristmas, setIsChristmas] = useState(false);
      const [position, setPosition] = useState({ x: 20, y: 20 });
      const [isDragging, setIsDragging] = useState(false);
      const [dragOffset, setDragOffset] = useState({ x: 0, y: 0 });
      const [emojiPositions, setEmojiPositions] = useState([]);
      const [lightColors, setLightColors] = useState([]);

      const emojis = ['ğŸ„', 'ğŸ', 'ğŸŒŸ', 'ğŸ…ğŸ½'];
      
      // Generate random Christmas light colors
      useEffect(() => {
        const colors = ['#ff0000', '#00ff00', '#0000ff', '#ffff00', '#ff00ff', '#00ffff', '#ff8800', '#ffffff'];
        const lights = Array(20).fill(0).map(() => ({
          color: colors[Math.floor(Math.random() * colors.length)],
          delay: Math.random() * 2
        }));
        setLightColors(lights);
      }, []);

      useEffect(() => {
        const calculateTime = () => {
          const now = new Date();
          const christmas = new Date(now.getFullYear(), 11, 25, 0, 0, 0);
          
          if (now >= christmas) {
            setIsChristmas(true);
            return { hours: 0, minutes: 0, seconds: 0 };
          }

          const diff = christmas - now;
          const hours = Math.floor(diff / (1000 * 60 * 60));
          const minutes = Math.floor((diff % (1000 * 60 * 60)) / (1000 * 60));
          const seconds = Math.floor((diff % (1000 * 60)) / 1000);

          return { hours, minutes, seconds };
        };

        const generateEmojiPositions = () => {
          return emojis.map((emoji, i) => ({
            emoji,
            x: 50 + Math.cos((Date.now() / 500 + i * Math.PI / 2)) * 60,
            y: 50 + Math.sin((Date.now() / 500 + i * Math.PI / 2)) * 60,
            rotate: (Date.now() / 50 + i * 90) % 360
          }));
        };

        const interval = setInterval(() => {
          setTime(calculateTime());
          if (isChristmas) {
            setEmojiPositions(generateEmojiPositions());
          }
        }, isChristmas ? 50 : 1000);

        return () => clearInterval(interval);
      }, [isChristmas]);

      const handleMouseDown = (e) => {
        setIsDragging(true);
        setDragOffset({
          x: e.clientX - position.x,
          y: e.clientY - position.y
        });
      };

      const handleMouseMove = (e) => {
        if (isDragging) {
          setPosition({
            x: e.clientX - dragOffset.x,
            y: e.clientY - dragOffset.y
          });
        }
      };

      const handleMouseUp = () => {
        setIsDragging(false);
      };

      useEffect(() => {
        if (isDragging) {
          window.addEventListener('mousemove', handleMouseMove);
          window.addEventListener('mouseup', handleMouseUp);
          return () => {
            window.removeEventListener('mousemove', handleMouseMove);
            window.removeEventListener('mouseup', handleMouseUp);
          };
        }
      }, [isDragging, dragOffset]);

      const GripVertical = () => (

        </svg>
      );

      return (

                  }}
                />
              ))}

                  className="absolute rounded-full"

                    height: '20px',

                    borderRadius: '50% 50% 45% 45%'
                  }}
                />
              ))}
              
              {/* Bottom Lights */}
              {lightColors.slice(10, 15).map((light, i) => (
                <div
                  key={`bottom-${i}`}
                  className="absolute rounded-full"
                  style={{
                    width: '16px',
                    height: '20px',
                    left: `${20 + i * 20}%`,
                    bottom: '-2px',
                    backgroundColor: light.color,
                    boxShadow: `0 0 10px ${light.color}, 0 0 20px ${light.color}`,
                    animation: `twinkle 1.5s ease-in-out ${light.delay}s infinite`,
                    borderRadius: '50% 50% 45% 45%'
                  }}
                />
              ))}
              
              {/* Left Lights */}
              {lightColors.slice(15, 20).map((light, i) => (
                <div
                  key={`left-${i}`}
                  className="absolute rounded-full"
                  style={{
                    width: '16px',
                    height: '20px',
                    left: '-2px',
                    top: `${20 + i * 20}%`,
                    backgroundColor: light.color,
                    boxShadow: `0 0 10px ${light.color}, 0 0 20px ${light.color}`,
                    animation: `twinkle 1.5s ease-in-out ${light.delay}s infinite`,
                    borderRadius: '50% 50% 45% 45%'
                  }}
                />
              ))}
              
              {/* Wire connecting lights */}
              <svg className="absolute inset-0 w-full h-full" style={{ zIndex: -1 }}>
                <path d="M 10,10 L 90,10 L 90,90 L 10,90 Z" stroke="#222" strokeWidth="2" fill="none" />
              </svg>
            </div>
            
            <div className="bg-gray-900 rounded-xl p-4 min-w-[280px] backdrop-blur-sm relative z-20">
              {/* Drag Handle */}
              <div className="flex justify-center mb-2 opacity-50 hover:opacity-100 transition-opacity">
                <GripVertical />
              </div>

              {!isChristmas ? (
                /* Countdown Display */
                <div className="text-center">
                  <div className="text-white text-sm font-semibold mb-2 tracking-wide">
                    ğŸ„ CHRISTMAS EVE ğŸ„

                      <div className="text-3xl font-bold text-white">{String(time.hours).padStart(2, '0')}</div>
                      <div className="text-xs text-red-200 uppercase">Hours</div>
                    </div>
                    <div className="bg-green-600 rounded-lg p-3 min-w-[60px]">
                      <div className="text-3xl font-bold text-white">{String(time.minutes).padStart(2, '0')}</div>
                      <div className="text-xs text-green-200 uppercase">Mins</div>
                    </div>
                    <div className="bg-red-600 rounded-lg p-3 min-w-[60px]">
                      <div className="text-3xl font-bold text-white">{String(time.seconds).padStart(2, '0')}</div>
                      <div className="text-xs text-red-200 uppercase">Secs</div>
                    </div>
                  </div>
                  <div className="text-yellow-300 text-xs animate-pulse">
                    Until Christmas! ğŸ…ğŸ½

                </div>
              ) : (
                /* Christmas Day Display */
                <div className="relative h-[200px] overflow-hidden">
                  {/* Dancing Emojis */}
                  <div className="absolute inset-0">
                    {emojiPositions.map((pos, i) => (
                      <div
                        key={i}
                        className="absolute text-4xl transition-all duration-100"
                        style={{
                          left: `${pos.x}%`,
                          top: `${pos.y}%`,
                          transform: `translate(-50%, -50%) rotate(${pos.rotate}deg)`,
                        }}
                      >
                        {pos.emoji}
                      </div>
                    ))}
                  </div>

                  {/* Center Time Display */}
                  <div className="absolute inset-0 flex items-center justify-center">
                    <div className="bg-black bg-opacity-50 rounded-xl p-4 backdrop-blur-sm">
                      <div className="text-5xl font-bold text-white mb-2 animate-pulse">
                        {new Date().toLocaleTimeString('en-US', { 
                          hour: '2-digit',

                      </div>

                        ğŸ¥·ğŸ½ RENT IS DUE IN 8 DAYS!! ğŸ«°ğŸ½ &nbsp;&nbsp;&nbsp;
                      </span>
                    </div>
                  </div>
                </div>
              )}
            </div>
          </div>
        </div>
      );
    }

    const root = ReactDOM.createRoot(document.getElementById('root'));
